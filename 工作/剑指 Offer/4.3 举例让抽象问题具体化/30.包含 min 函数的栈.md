定义栈的数据结构，请在该类型中实现一个能够得到栈的最小元素的 min 函数在该栈中，调用 min、push 及 pop 的时间复杂度都是 O(1)。

```
示例:

MinStack minStack = new MinStack();
minStack.push(-2);
minStack.push(0);
minStack.push(-3);
minStack.min();   --> 返回 -3.
minStack.pop();
minStack.top();      --> 返回 0.
minStack.min();   --> 返回 -2.
```


提示：

各函数的调用总次数不超过 20000 次

---

方法：（使用一个辅助的递减栈来记录栈中的最小值，即栈顶就是最小值）

```javascript
/**
 * initialize your data structure here.
 */
var MinStack = function() {
  this.minStack = [];
  this.minStackTemp = [];
};

/** 
 * @param {number} x
 * @return {void}
 */
MinStack.prototype.push = function(x) {
  this.minStack.push(x);
  if (this.minStackTemp.length === 0) {
    this.minStackTemp.push(x);
  } else if (this.minStackTemp[this.minStackTemp.length - 1] >= x) {
    this.minStackTemp.push(x);
  }
};

/**
 * @return {void}
 */
MinStack.prototype.pop = function() {
  var x = this.minStack.pop();
  if (this.minStackTemp[this.minStackTemp.length - 1] === x) {
    this.minStackTemp.pop();
  }
};

/**
 * @return {number}
 */
MinStack.prototype.top = function() {
  if (this.minStack.length === 0) return null;
  return this.minStack[this.minStack.length - 1];
};

/**
 * @return {number}
 */
MinStack.prototype.min = function() {
  if (this.minStack.length === 0) return null;
  return this.minStackTemp[this.minStackTemp.length - 1];
};

/**
 * Your MinStack object will be instantiated and called as such:
 * var obj = new MinStack()
 * obj.push(x)
 * obj.pop()
 * var param_3 = obj.top()
 * var param_4 = obj.min()
 */
```
