输入两个递增排序的链表，合并这两个链表并使新链表中的节点仍然是递增排序的。

```
示例1：

输入：1->2->4, 1->3->4
输出：1->1->2->3->4->4
```


限制：

0 <= 链表长度 <= 1000

---

方法：迭代法

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var mergeTwoLists = function(l1, l2) {
  if (l1 && !l2) {
    return l1
  } else if (!l1 && l2) {
    return l2
  } else if (!l1 && !l2) {
    return null
  } else if (l1 && l2) {
    let p = l1;
    let j = l2;
    let head = null;
    let res = null;
    while (p && j) {
      if (p.val <= j.val) {
        if (head === null) {
          head = new ListNode(p.val);
          res = head;
        } else {
          res.next = new ListNode(p.val);
          res = res.next;
        }
        p = p.next;
      } else if (p.val > j.val) {
        if (head === null) {
          head = new ListNode(j.val);
          res = head;
        } else {
          res.next = new ListNode(j.val);
          res = res.next
        }
        j = j.next;
      }
    }
    if(p) {
      res.next = p
    }
    if(j) {
      res.next = j
    }
    return head;
  }
  
};

// var l1 = { val: 1, next: {val: 2, next: null} }

```
